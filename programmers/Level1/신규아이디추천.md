### :heavy_check_mark: 내가 푼 답안

```javascript
function solution(new_id) {
    let arr = [];
    
    arr = [...[...new_id.toLowerCase()].map(e => /^[a-zA-Z0-9-_.]*$/.test(e) ? e : '').join('')]; // 1, 2 
    
    for(let i=0;i<arr.length;i++){ // 3
        if(arr[i] == '.' && arr[i] == arr[i+1]){
            arr.splice(i,1);
            i--;
        }
    }
    
    if(arr[0] =='.') arr.splice(0,1); // 4
    if(arr[arr.length-1] =='.') arr.splice(arr.length-1,1);
    
    if(arr.join('') == ''){ arr[0] = 'a'} // 5
    
    if(arr.length >= 16){ arr.splice(15,arr.length-15);} //6
    if(arr[arr.length-1] =='.') arr.splice(arr.length-1,1); 

    if(arr.length <= 2 ){ //7
        const len = arr.length;
        while(arr.length < 3){
            arr.push(arr[len-1]);
        }
    }
    
    return arr.join('');
}
```
한 단계씩 무슨 퀴즈를 푸는 기분이였다.  
정규식은 아직 잘 몰라서 검색을 했다.. 나중에 공부해야겠다..

```javascript
function solution(new_id) {
    const answer = new_id
        .toLowerCase() // 1
        .replace(/[^\w-_.]/g, '') // 2
        .replace(/\.+/g, '.') // 3
        .replace(/^\.|\.$/g, '') // 4
        .replace(/^$/, 'a') // 5
        .slice(0, 15).replace(/\.$/, ''); // 6
    const len = answer.length;
    return len > 2 ? answer : answer + answer.charAt(len - 1).repeat(3 - len);
}
```
와우 체이닝을 이용해서.. replace함수를 쓰면 간편해지는 코드였다.  
정규식의 필요성을 느낀다..  
repeat 함수 계속 보기만하고 생각을 못한다.. 잊지말자.

```javascript
function solution(nid) {
  var ans = "";
  for (let i = 0; i < nid.length; i++) {
    let c = nid[i].toLowerCase();
    if ("0123456789abcdefghijklmnopqrstuvwxyz.-_".indexOf(c) === -1) continue;
    if (c === "." && ans[ans.length - 1] === "." && nid[i - 1]) continue;
    ans += c;
  }
  if (ans[0] === ".") ans = ans.slice(1);
  ans = ans.slice(0, 15);
  if (ans[ans.length - 1] === ".") ans = ans.slice(0, ans.length - 1);
  if (!ans) ans = "a";
  while (ans.length < 3) ans += ans[ans.length - 1];
  return ans;
}
```
