```javascript
function solution(s) {
  let count = 0,
    zeroCount = 0;
  let preNum = 0,
    zeroNum = 0;

  while (true) {
    if (s.length == 1) {
      return [count, zeroCount];
    }

    preNum = s.length;
```

```javascript
function solution(s) {
  var answer = [0, 0];
  while (s.length > 1) {
    answer[0]++;
    answer[1] += (s.match(/0/g) || []).length;
    s = s.replace(/0/g, '').length.toString(2);
  }
  return answer;
}
```

따로 변수를 주지않고 배열에서 ++ 해줬다.
