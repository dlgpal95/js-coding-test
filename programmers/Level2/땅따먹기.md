### :heavy_check_mark: 내가 푼 답안

```javascript
function solution(land) {
  let answer = 0;
  let MaxNum = 0,
    MaxIndex = 0;

  land.map((e, i) => {
    if (i !== 0) e.splice(MaxIndex, 1);
    e.map((el, j) => {
      if (MaxNum <= el) {
        MaxIndex = j;
        MaxNum = el;
      }
    });
    answer += MaxNum;
    MaxNum = 0;
  });

  return answer;
}
```

하면 TestCase는 통과가 된다. 근데 제출에서 다 실패를 한다.  
각 행에 가장 큰값을 더하면 그게 최대값이 될 줄 알았는데 아니였다.  
결과적으로 큰 수가 되어야해서

[[1,2,3,5]  
[5,6,7,100]  
[4,3,2,1]]

이 경우 첫번째 행에 5를 선택하지말고 3을 선택해서 두번째 행에 100을 선택할수 있게해야  
최대값이 된다.

```javascript
function solution(land) {
  for (let i = land.length - 2; i >= 0; i--) {
    land[i][0] = Math.max(land[i + 1][1], land[i + 1][2], land[i + 1][3]) + land[i][0];
    land[i][1] = Math.max(land[i + 1][0], land[i + 1][2], land[i + 1][3]) + land[i][1];
    land[i][2] = Math.max(land[i + 1][0], land[i + 1][1], land[i + 1][3]) + land[i][2];
    land[i][3] = Math.max(land[i + 1][0], land[i + 1][1], land[i + 1][2]) + land[i][3];
  }
  let answer = Math.max(...land[0]);

  return answer;
}
```
